public void NestedTryCatch()
{
	try
	{
		Console.WriteLine(1);
		try
		{
			Console.WriteLine(2);
		}
		catch
		{
			Console.WriteLine(3);
		}
		Console.WriteLine(4);
	}
	catch
	{
		try
		{
			Console.WriteLine(5);
		}
		catch
		{
			Console.WriteLine(6);
		}
	}
	Console.WriteLine(7);
}
